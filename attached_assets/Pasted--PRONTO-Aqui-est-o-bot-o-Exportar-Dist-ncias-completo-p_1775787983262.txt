// ====================== BOTÃO EXPORTAR DISTÂNCIAS ======================
// Cole este componente no final do seu arquivo principal do atlas

import { useState } from 'react';

interface Picto {
  id: string | number;
  palavra: string;
  categoria: string;
  coordX: number;
  coordY: number;
  imagemUrl?: string;
}

const ButtonExportDistances = ({ pictos }: { pictos: Picto[] }) => {
  const [isExporting, setIsExporting] = useState(false);

  const exportDistances = () => {
    if (!pictos || pictos.length === 0) {
      alert('❌ Nenhum pictograma carregado!');
      return;
    }

    setIsExporting(true);
    console.log(`🚀 Exportando distâncias de ${pictos.length} pictogramas...`);

    // Calcular matriz de distâncias euclidianas (igual ao cloud)
    const coords = pictos.map(p => [p.coordX, p.coordY]);

    function euclidean(a: number[], b: number[]) {
      return Math.hypot(a[0] - b[0], a[1] - b[1]);
    }

    const distMatrix = coords.map((c1, i) =>
      coords.map((c2, j) => i === j ? 0 : parseFloat(euclidean(c1, c2).toFixed(6)))
    );

    const exportData = {
      metadata: {
        total_pictos: pictos.length,
        algorithm: "Wasserstein + MDS (Euclidean projection)",
        generated_at: new Date().toISOString(),
        author: "João Pedro Pereira Passos - AFASIA IAP",
        version: "1.0"
      },
      palavras: pictos.map(p => p.palavra),
      dist_matrix: distMatrix,
      pictos: pictos.map(p => ({
        id: p.id,
        palavra: p.palavra,
        categoria: p.categoria,
        coordX: p.coordX,
        coordY: p.coordY,
        imagemUrl: p.imagemUrl
      }))
    };

    // Download automático
    const blob = new Blob([JSON.stringify(exportData, null, 2)], { type: 'application/json' });
    const url = URL.createObjectURL(blob);
    const link = document.createElement('a');
    link.href = url;
    link.download = `atlas_distances_full_${pictos.length}_icones.json`;
    link.click();

    URL.revokeObjectURL(url);
    setIsExporting(false);

    alert(`✅ Download concluído!\n\nArquivo: atlas_distances_full_${pictos.length}_icones.json\n\nAgora copie para o repositório GitHub AFASIA!`);
  };

  return (
    
      {isExporting ? '⏳ Exportando...' : '📤 Exportar Todas as Distâncias'}
    
  );
};

export default ButtonExportDistances;
