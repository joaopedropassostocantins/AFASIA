// ================================================
// PROMPT DETERMINADO - EXPORTAR DISTÂNCIAS + PREPARAR PARA REPOSITÓRIO
// Cole este bloco inteiro no Console do Replit
// ================================================

console.log("%c🚀 Iniciando exportação completa das distâncias...", "color:#00ff00; font-size:16px; font-weight:bold");

const pictos = window.__ATLAS_PICTOS || 
               (window.appState && window.appState.pictos) || 
               (document.querySelector('[data-testid="atlas-cloud"]') && 
                document.querySelector('[data-testid="atlas-cloud"]').__reactFiber?.memoizedState?.baseState?.pictos);

if (!pictos || !Array.isArray(pictos) || pictos.length === 0) {
    alert("❌ Não consegui encontrar os dados automaticamente.\n\n1. Instale a extensão React DevTools\n2. Abra a aba Components\n3. Procure o componente que contém 'pictos'\n4. Expanda e rode o script novamente.");
    throw new Error("Dados não encontrados");
}

console.log(`✅ Encontrados ${pictos.length} pictogramas!`);

// Extrair coordenadas e palavras
const coords = pictos.map(p => [parseFloat(p.coordX), parseFloat(p.coordY)]);
const palavras = pictos.map(p => p.palavra);

// Calcular matriz de distâncias (exatamente como o app usa)
function euclidean(a, b) {
    return Math.hypot(a[0] - b[0], a[1] - b[1]);
}

const distMatrix = coords.map((c1, i) => 
    coords.map((c2, j) => i === j ? 0 : parseFloat(euclidean(c1, c2).toFixed(6)))
);

const exportData = {
    metadata: {
        total_pictos: pictos.length,
        algorithm: "Wasserstein + MDS (Euclidean projection)",
        generated_at: new Date().toISOString(),
        author: "João Pedro Pereira Passos"
    },
    palavras: palavras,
    dist_matrix: distMatrix,
    pictos: pictos.map(p => ({
        id: p.id,
        palavra: p.palavra,
        categoria: p.categoria,
        coordX: p.coordX,
        coordY: p.coordY,
        imagemUrl: p.imagemUrl
    }))
};

// Baixar o arquivo automaticamente
const blob = new Blob([JSON.stringify(exportData, null, 2)], { type: "application/json" });
const url = URL.createObjectURL(blob);
const a = document.createElement("a");
a.href = url;
a.download = `atlas_distances_full_${pictos.length}_icones.json`;
a.click();

console.log("%c✅ ARQUIVO BAIXADO COM SUCESSO!", "color:#00ff00; font-size:18px");
console.log("Nome do arquivo:", a.download);
console.log("Agora copie este arquivo para a pasta do seu repositório GitHub.");
