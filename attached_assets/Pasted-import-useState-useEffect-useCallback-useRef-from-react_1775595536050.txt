// ── API ANTHROPIC (simula Gemma 4) ────────────────────────────────
async function inferirComIA(pictogramas, historico) {
const pics = pictogramas.map(p => `${p.label} (${p.cat})`).join(”, “);
const hist = historico.slice(-3).join(” | “) || “nenhum”;

const prompt = `Voce e um sistema de CAA (Comunicacao Aumentativa e Alternativa) para pessoas com afasia.

Pictogramas selecionados: [${pics}]
Periodo do dia: ${getPeriodo()}
Historico recente: ${hist}

Responda SOMENTE com JSON valido, sem texto fora do JSON:
{
“intencao”: “frase natural em portugues brasileiro simples e direta”,
“urgencia”: 0,
“emocao”: “neutro|alegre|ansioso|com_dor|com_medo”,
“confianca”: 0.0,
“sugestoes”: [“id1”, “id2”, “id3”],
“nota_cuidador”: “orientacao breve para quem assiste”
}

Ids validos para sugestoes: agua, comida, banheiro, remedio, descanso, frio, calor, dor, bem, mal, ansioso, feliz, triste, com_medo, casa, hospital, quarto, jardim, sala, filho, medico, enfermeiro, familia, tv, telefone, musica, ler, sair`;

try {
const res = await fetch(“https://api.anthropic.com/v1/messages”, {
method: “POST”,
headers: { “Content-Type”: “application/json” },
body: JSON.stringify({
model: “claude-sonnet-4-20250514”,
max_tokens: 300,
messages: [{ role: “user”, content: prompt }],
}),
});
const data = await res.json();
const texto = data.content?.[0]?.text || “”;
const ini = texto.indexOf(”{”);
const fim = texto.lastIndexOf(”}”) + 1;
return JSON.parse(texto.slice(ini, fim));
} catch {
return null;
}
}
